const Database = require('better-sqlite3');

  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,
  description TEXT,
  price REAL NOT NULL
);
`);

// Insert products
const insert = db.prepare('INSERT INTO products (name, description, price) VALUES (?, ?, ?)');
const insertAll = db.transaction((rows) => {
  for (const r of rows) insert.run(...r);

TASK 3 – Vulnerable Functions & Attacks
========================================

-- vulnerable.js code --

const Database = require('better-sqlite3');
const db = new Database('nyondo_stock.db');

// Ensure users table exists
db.exec(`
CREATE TABLE IF NOT EXISTS users (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  username TEXT NOT NULL,
  password TEXT NOT NULL,
  role TEXT DEFAULT 'attendant'
);
`);

db.exec(`
INSERT OR IGNORE INTO users (username, password, role) VALUES
('admin', 'admin123', 'admin'),
('fatuma', 'pass456', 'attendant'),
('wasswa', 'pass789', 'manager');
`);

function searchProduct(name) {
  const query = `SELECT * FROM products WHERE name LIKE '${name}'`;
  console.log("Query:", query);
  const rows = db.prepare(query).all();
  console.log("Result:", rows, "\n");
  return rows;
}

function login(username, password) {
  const query = `SELECT * FROM users WHERE username='${username}' AND password='${password}'`;
  console.log("Query:", query);
  const row = db.prepare(query).get();
  console.log("Result:", row, "\n");
  return row;
}

// Attacks
searchProduct("' OR 1=1 -- ");
login("admin' -- ", "anything");
login("' OR '1'='1", "' OR '1'='1");
searchProduct("' UNION SELECT id, username, password, role FROM users -- ");


-- Attack 1: ' OR 1=1 -- --
Query:
SELECT * FROM products WHERE name LIKE '' OR 1=1 -- '
Result:
[All products returned]

Explanation:
The condition `OR 1=1` makes the WHERE clause always true, so the query ignores the filter and dumps all products.

-- Attack 2: ' -- --
Query:
SELECT * FROM users WHERE username='admin' -- ' AND password='anything'
Result:
{ id: 1, username: 'admin', password: 'admin123', role: 'admin' }

Explanation:
The `--` comment cancels the password check, allowing login as admin without verifying the password.

-- Attack 3: ' OR '1'='1 --
Query:
SELECT * FROM users WHERE username='' OR '1'='1' AND password='' OR '1'='1'
Result:
{ id: 1, username: 'admin', password: 'admin123', role: 'admin' }

Explanation:
The condition `'1'='1'` is always true, so the query authenticates the user without checking credentials.

-- Attack 4: UNION SELECT --
Query:
SELECT * FROM products WHERE name LIKE '' UNION SELECT id, username, password, role FROM users -- '
Result:
[
  { id: 1, name: 'admin', description: 'admin123', price: 'admin' },
  { id: 2, name: 'fatuma', description: 'pass456', price: 'attendant' },
  { id: 3, name: 'wasswa', description: 'pass789', price: 'manager' }
]

Explanation:
The UNION merges the users table into the product search, exposing usernames and passwords.

TASK 4 – Secure Version
========================================

-- secure.js code --
const Database = require('better-sqlite3');
const db = new Database('nyondo_stock.db');

function searchProduct(name) {
  const rows = db.prepare("SELECT * FROM products WHERE name LIKE ?").all(name);
  console.log("Secure product search result:", rows);
}

function login(username, password) {
  const row = db.prepare("SELECT * FROM users WHERE username=? AND password=?").get(username, password);
  console.log("Secure login result:", row);
}

// Test with malicious inputs
searchProduct("' OR 1=1 --");
login("admin' --", "anything");
login("' OR '1'='1", "' OR '1'='1");
searchProduct("' UNION SELECT id, username, password, role FROM users --");

-- Terminal outputs --
Secure product search result: []
Secure login result: undefined
Secure login result: undefined
Secure product search result: []

-- Explanation --
Using `?` placeholders prevents SQL injection because user input is treated strictly as data, not executable SQL. Unlike string concatenation, parameterized queries separate query logic from input, so malicious payloads cannot alter the query structure. This is why all attack attempts fail and return empty results.

-- validation.js code --
function validateUsername(username) {
  // Reject empty or too short usernames
  if (!username || username.length < 3) {
    return false;
  }
  // Reject usernames with spaces or special characters
  const regex = /^[a-zA-Z0-9_]+$/;
  return regex.test(username);
}

function validatePassword(password) {
  // Require minimum length
  if (!password || password.length < 6) {
    return false;
  }
  // Reject obvious patterns
  if (password.toLowerCase() === "password") {
    return false;
  }
  return true;
}

function validateProductName(name) {
  // Reject dangerous characters often used in SQL injection
  const forbidden = /['";]/;   // ✅ removed the problematic -- 
  return !forbidden.test(name);
}

// Example tests
console.log("Username test:", validateUsername("ad"));       // false
console.log("Password test:", validatePassword("123"));      // false
console.log("Product test:", validateProductName("Cement")); // true
console.log("Product test:", validateProductName("' OR 1=1")); // false

-- Terminal outputs --
Username test: false
Password test: false
Product test: true
Product test: false

-- Explanation --
Parameterization protects the database, but input validation protects the application itself. By rejecting invalid inputs (like too short passwords, usernames with spaces, or dangerous characters in product names), the system ensures only clean, expected data reaches the database. This adds another layer of security beyond SQL injection prevention.
